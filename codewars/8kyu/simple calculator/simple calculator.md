### Switch 

```js
function calculator(a,b,sign){
	
//	if( !isFinite(a) || !isFinite(b) ) return 'number Invalid operator'; ‚≠ï
  if (typeof a !== 'number' || typeof b !== 'number') return 'no number';

  switch(sign) {
      
	  case '/':
		  return  a / b;

	  case '*':
		   return a * b;
	
	  case '+':
		   return a + b;
	
	  case '-':
		   return a - b;
		
	  default: 
		  return 'Invalid operator';  
  }
}
```
